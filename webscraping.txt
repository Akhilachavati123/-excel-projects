from bs4 import BeautifulSoup
import requests

url ='https://www.scrapethissite.com/pages/forms/'

page = requests.get(url)

print(page)

soup = BeautifulSoup(page.text, 'html')

print(soup)

soup.find_all('table')

soup.find_all('table', class_ = 'table')

table = soup.find_all('table')

print(table)

hockey_titles = soup.find_all('th')

hockey_titles

hockey_table_titles = [title.text.strip() for title in hockey_titles]

print(hockey_table_titles)

import pandas as pd

df = pd.DataFrame(columns = hockey_table_titles)

df

column_data = soup.find_all('tr')

for row in column_data[1:]:
    row_data = row.find_all('td')
    individual_row_data = [data.text.strip() for data in row_data]

    length = len(df)
    df.loc[length] = individual_row_data


df

df.to_csv(r'C:\Users\91807\OneDrive\Attachments\Web scraping project\Hockey_teams.csv', index = False)
